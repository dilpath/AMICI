97 void NewtonSolver::computeNewtonSensis(AmiVectorArray &sx) {
98  prepareLinearSystem(0, -1);
99  model->fdxdotdp(*t, *x, dx);
100 
101  if (model->pythonGenerated) {
102  for (int ip = 0; ip < model->nplist(); ip++) {
103  N_VConst(0.0, sx.getNVector(ip));
104 
105  // copy explicit version
106  if (model->ndxdotdp_explicit > 0) {
107  auto col = model->dxdotdp_explicit.indexptrs();
108  auto row = model->dxdotdp_explicit.indexvals();
109  auto data_ptr = model->dxdotdp_explicit.data();
110  for (sunindextype iCol = col[model->plist(ip)];
111  iCol < col[model->plist(ip) + 1]; ++iCol)
112  sx.at(row[iCol], ip) -= data_ptr[iCol];
113  }
114 
115  // copy implicit version
116  if (model->ndxdotdp_implicit > 0) {
117  auto col = model->dxdotdp_implicit.indexptrs();
118  auto row = model->dxdotdp_implicit.indexvals();
119  auto data_ptr = model->dxdotdp_implicit.data();
120  for (sunindextype iCol = col[model->plist(ip)];
121  iCol < col[model->plist(ip) + 1]; ++iCol)
122  sx.at(row[iCol], ip) -= data_ptr[iCol];
123  }
124 
125  solveLinearSystem(sx[ip]);
126  }
127  } else {
128  for (int ip = 0; ip < model->nplist(); ip++) {
129  for (int ix = 0; ix < model->nx_solver; ix++)
130  sx.at(ix,ip) = -model->dxdotdp.at(ix, ip);
131 
132  solveLinearSystem(sx[ip]);
133  }
134  }
135 }

258 void NewtonSolverIterative::linsolveSPBCG(int ntry, int nnewt,
259  AmiVector &ns_delta) {
260  xdot = ns_delta;
261  xdot.minus();
262 
263  // Get the diagonal of the Jacobian for preconditioning
264  model->fJDiag(*t, ns_Jdiag, 0.0, *x, dx);
265 
266  // Ensure positivity of entries in ns_Jdiag
267  ns_p.set(1.0);
268  N_VAbs(ns_Jdiag.getNVector(), ns_Jdiag.getNVector());
269  N_VCompare(1e-15, ns_Jdiag.getNVector(), ns_tmp.getNVector());
270  N_VLinearSum(-1.0, ns_tmp.getNVector(), 1.0, ns_p.getNVector(), ns_tmp.getNVector());
271  N_VLinearSum(1.0, ns_Jdiag.getNVector(), 1.0, ns_tmp.getNVector(), ns_Jdiag.getNVector());
272 
273  // Initialize for linear solve
274  ns_p.reset();
275  ns_v.reset();
276  ns_delta.reset();
277  ns_tmp.reset();
278  double rho = 1.0;
279  double omega = 1.0;
280  double alpha = 1.0;
281 
282  // can be set to 0 at the moment
283  model->fJv(*t, *x, dx, xdot, ns_delta, ns_Jv, 0.0);
284 
285  // ns_r = xdot - ns_Jv;
286  N_VLinearSum(-1.0, ns_Jv.getNVector(), 1.0, xdot.getNVector(), ns_r.getNVector());
287  N_VDiv(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
288  ns_rt = ns_r;
289 
290  for (int i_linstep = 0; i_linstep < maxlinsteps;
291  i_linstep++) {
292  // Compute factors
293  double rho1 = rho;
294  rho = N_VDotProd(ns_rt.getNVector(), ns_r.getNVector());
295  double beta = rho * alpha / (rho1 * omega);
296 
297  // ns_p = ns_r + beta * (ns_p - omega * ns_v);
298  N_VLinearSum(1.0, ns_p.getNVector(), -omega, ns_v.getNVector(), ns_p.getNVector());
299  N_VLinearSum(1.0, ns_r.getNVector(), beta, ns_p.getNVector(), ns_p.getNVector());
300 
301  // ns_v = J * ns_p
302  model->fJv(*t, *x, dx, xdot, ns_p, ns_v, 0.0);
303  N_VDiv(ns_v.getNVector(), ns_Jdiag.getNVector(), ns_v.getNVector());
304 
305  // Compute factor
306  alpha = rho / N_VDotProd(ns_rt.getNVector(), ns_v.getNVector());
307 
308  // ns_h = ns_delta + alpha * ns_p;
309  N_VLinearSum(1.0, ns_delta.getNVector(), alpha, ns_p.getNVector(),
310  ns_h.getNVector());
311  // ns_s = ns_r - alpha * ns_v;
312  N_VLinearSum(1.0, ns_r.getNVector(), -alpha, ns_v.getNVector(), ns_s.getNVector());
313 
314  // ns_t = J * ns_s
315  model->fJv(*t, *x, dx, xdot, ns_s, ns_t, 0.0);
316  N_VDiv(ns_t.getNVector(), ns_Jdiag.getNVector(), ns_t.getNVector());
317 
318  // Compute factor
319  omega = N_VDotProd(ns_t.getNVector(), ns_s.getNVector()) / N_VDotProd(ns_t.getNVector(), ns_t.getNVector());
320 
321  // ns_delta = ns_h + omega * ns_s;
322  N_VLinearSum(1.0, ns_h.getNVector(), omega, ns_s.getNVector(),
323  ns_delta.getNVector());
324  // ns_r = ns_s - omega * ns_t;
325  N_VLinearSum(1.0, ns_s.getNVector(), -omega, ns_t.getNVector(), ns_r.getNVector());
326 
327  // Compute the (unscaled) residual
328  N_VProd(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
329  double res = sqrt(N_VDotProd(ns_r.getNVector(), ns_r.getNVector()));
330 
331  // Test convergence
332  if (res < atol) {
333  // Write number of steps needed
334  rdata->newton_numlinsteps[(ntry - 1) * maxsteps +
335  nnewt] = i_linstep + 1;
336 
337  // Return success
338  return;
339  }
340 
341  // Scale back
342  N_VDiv(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
343  }
344  throw NewtonFailure(AMICI_CONV_FAILURE, "linsolveSPBCG");
345 }
