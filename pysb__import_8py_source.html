594 def _compute_target_index(cl_prototypes, ODE):
595  possible_indices = list(set(list(itertools.chain(*[
596  cl_prototypes[monomer]['possible_indices']
597  for monomer in cl_prototypes
598  ]))))
599 
600  # Note: currently this function is supposed to also count appearances in
601  # expressions. However, expressions are currently still empty as they
602  # are also populated from conservation laws. In case there are many
603  # state heavy expressions in the model (should not be the case for mass
604  # action kinetics). This may lead to suboptimal results and could improved.
605  # As this would require substantial code shuffling, this will only be
606  # fixed if this becomes an actual problem
607  appearance_counts = ODE.get_appearance_counts(possible_indices)
608 
609  # in this initial guess we ignore the cost of having cyclic dependencies
610  # between conservation laws
611  for monomer in cl_prototypes:
612  prototype = cl_prototypes[monomer]
613  # extract monomer specific appearance counts
614  prototype['appearance_counts'] = \
615  [
616  appearance_counts[possible_indices.index(idx)]
617  for idx in prototype['possible_indices']
618  ]
619  # select target index as possible index with minimal appearance count
620  if len(prototype['appearance_counts']) == 0:
621  raise Exception(f'Failed to compute conservation law for monomer '
622  f'{monomer}')
623 
624  idx = np.argmin(prototype['appearance_counts'])
625 
626  # remove entries from possible indices and appearance counts so we
627  # do not consider them again in later iterations
628  prototype['target_index'] = prototype['possible_indices'].pop(idx)
629  prototype['appearance_count'] = prototype['appearance_counts'].pop(idx)
630 
631  # this is only an approximation as the effective species count
632  # of other conservation laws may also be affected by the chosen
633  # target index. As long as the number of unique monomers in
634  # multimers has a low upper bound and the species count does not
635  # vary too much across conservation laws, this approximation
636  # should be fine
637  prototype['fillin'] = \
638  prototype['appearance_count'] * prototype['species_count']
639 
640  # we might end up with the same index for multiple monomers, so loop until
641  # we have a set of unique target indices
642  while not _cl_prototypes_are_valid(cl_prototypes):
643  _greedy_target_index_update(cl_prototypes)

768 def _greedy_target_index_update(cl_prototypes):
769 
770  target_indices = get_target_indices(cl_prototypes)
771 
772  for monomer, prototype in cl_prototypes.items():
773  if target_indices.count(prototype['target_index']) > 1 or \
774  _cl_has_cycle(monomer, cl_prototypes):
775  # compute how much fillin the next best target_index would yield
776 
777  # we exclude already existing target indices to avoid that
778  # updating the target index removes uniqueness from already unique
779  # target indices, this may slightly reduce chances of finding a
780  # solution but prevents infinite loops
781  for target_index in list(set(target_indices)):
782  try:
783  local_idx = prototype['possible_indices'].index(
784  target_index
785  )
786  except ValueError:
787  local_idx = None
788 
789  if local_idx:
790  del prototype['possible_indices'][local_idx]
791  del prototype['appearance_counts'][local_idx]
792 
793  if len(prototype['possible_indices']) == 0:
794  prototype['diff_fillin'] = -1
795  continue
796 
797  idx = np.argmin(prototype['appearance_counts'])
798 
799  prototype['local_index'] = idx
800  prototype['alternate_target_index'] = \
801  prototype['possible_indices'][idx]
802  prototype['alternate_appearance_count'] = \
803  prototype['appearance_counts'][idx]
804 
805  prototype['alternate_fillin'] = \
806  prototype['alternate_appearance_count'] \
807  * prototype['species_count']
808 
809  prototype['diff_fillin'] = \
810  prototype['alternate_fillin'] - prototype['fillin']
811  else:
812  prototype['diff_fillin'] = -1
813 
814  if all(
815  prototype['diff_fillin'] == -1
816  for prototype in cl_prototypes.values()
817  ):
818  raise Exception('Could not compute a valid set of conservation '
819  'laws for this model!')
820 
821  # this puts prototypes with high diff_fillin last
822  cl_prototypes = sorted(
823  cl_prototypes.items(), key=lambda kv: kv[1]['diff_fillin']
824  )
825  cl_prototypes = {
826  proto[0]: proto[1]
827  for proto in cl_prototypes
828  }
829 
830  for monomer in cl_prototypes:
831  prototype = cl_prototypes[monomer]
832  # we check that we
833  # A) have an alternative index computed, i.e. that
834  # that monomer originally had a non-unique target_index
835  # B) that the target_index still is not unique or part of a cyclic
836  # dependency. due to the sorting, this will always be the monomer
837  # with the highest diff_fillin (note that the target index counts
838  # are recomputed on the fly)
839 
840  if prototype['diff_fillin'] > -1 \
841  and (
842  get_target_indices(cl_prototypes).count(
843  prototype['target_index']
844  ) > 1
845  or _cl_has_cycle(monomer, cl_prototypes)
846  ):
847  prototype['fillin'] = prototype['alternate_fillin']
848  prototype['target_index'] = prototype['alternate_target_index']
849  prototype['appearance_count'] = \
850  prototype['alternate_appearance_count']
851 
852  del prototype['possible_indices'][prototype['local_index']]
853  del prototype['appearance_counts'][prototype['local_index']]

889 def _construct_conservation_from_prototypes(cl_prototypes, model):
890  conservation_laws = []
891  for monomer_name in cl_prototypes:
892  target_index = cl_prototypes[monomer_name]['target_index']
893 
894  # T = sum_i(a_i * x_i)
895  # x_j = (T - sum_i≠j(a_i * x_i))/a_j
896  # law: sum_i≠j(a_i * x_i))/a_j
897  # state: x_j
898  target_expression = sum(
899  sp.Symbol(f'__s{ix}')
900  * extract_monomers(specie).count(monomer_name)
901  for ix, specie in enumerate(model.species)
902  if ix != target_index
903  ) / extract_monomers(model.species[target_index]).count(monomer_name)
904  # normalize by the stoichiometry of the target species
905  target_state = sp.Symbol(f'__s{target_index}')
906  # = x_j
907 
908  total_abundance = sp.Symbol(f'tcl__s{target_index}')
909  # = T/a_j
910 
911  state_expr = total_abundance - target_expression
912  # x_j = T/a_j - sum_i≠j(a_i * x_i)/a_j
913 
914  abundance_expr = target_expression + target_state
915  # T/a_j = sum_i≠j(a_i * x_i)/a_j + x_j
916 
917  conservation_laws.append({
918  'state': target_state,
919  'total_abundance': total_abundance,
920  'state_expr': state_expr,
921  'abundance_expr': abundance_expr,
922  })
923 
924  return conservation_laws
