68 void Solver::setup(const realtype t0, Model *model, const AmiVector &x0,
69  const AmiVector &dx0, const AmiVectorArray &sx0,
70  const AmiVectorArray &sdx0) const {
71  if (nx() != model->nx_solver || nplist() != model->nplist() ||
72  nquad() != model->nJ * model->nplist()) {
73  resetMutableMemory(model->nx_solver, model->nplist(),
74  model->nJ * model->nplist());
75  }
76  /* Create solver memory object if necessary */
77  allocateSolver();
78  if (!solverMemory)
79  throw AmiException("Failed to allocated solver memory!");
80 
81  /* Initialize CVodes/IDAs solver*/
82  init(t0, x0, dx0);
83 
84  applyTolerances();
85 
86  /* Set optional inputs */
87  setErrHandlerFn();
88  /* attaches userdata*/
89  setUserData(model);
90  /* specify maximal number of steps */
91  setMaxNumSteps(maxsteps);
92  /* activates stability limit detection */
93  setStabLimDet(stldet);
94 
95  rootInit(model->ne);
96 
97  initializeLinearSolver(model);
98  initializeNonLinearSolver();
99 
100  if (sensi >= SensitivityOrder::first && model->nx_solver > 0) {
101  auto plist = model->getParameterList();
102  if (sensi_meth == SensitivityMethod::forward && !plist.empty()) {
103  /* Set sensitivity analysis optional inputs */
104  auto par = model->getUnscaledParameters();
105 
106  /* Activate sensitivity calculations */
107  sensInit1(sx0, sdx0);
108  initializeNonLinearSolverSens(model);
109  setSensParams(par.data(), nullptr, plist.data());
110 
111  applyTolerancesFSA();
112  } else if (sensi_meth == SensitivityMethod::adjoint) {
113  /* Allocate space for the adjoint computation */
114  adjInit();
115  }
116  }
117 
118  setId(model);
119  setSuppressAlg(true);
120  /* calculate consistent DAE initial conditions (no effect for ODE) */
121  if (model->nt() > 1)
122  calcIC(model->getTimepoint(1));
123 }
