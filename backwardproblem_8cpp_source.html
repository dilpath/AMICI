36 void BackwardProblem::workBackwardProblem() {
37 
38 
39  if (model->nx_solver <= 0 ||
40  solver->getSensitivityOrder() < SensitivityOrder::first ||
41  solver->getSensitivityMethod() != SensitivityMethod::adjoint ||
42  model->nplist() == 0) {
43  return;
44  }
45 
46  int it = rdata->nt - 1;
47  model->initializeB(xB, dxB, xQB);
48  handleDataPointB(it);
49  solver->setupB(&which, rdata->ts[it], model, xB, dxB, xQB);
50 
51  --it;
52  --iroot;
53 
54  while (it >= 0 || iroot >= 0) {
55 
56  /* check if next timepoint is a discontinuity or a data-point */
57  double tnext = getTnext(discs, iroot, it);
58 
59  if (tnext < t) {
60  solver->runB(tnext);
61  solver->writeSolutionB(&t, xB, dxB, xQB, this->which);
62  solver->getDiagnosisB(it, rdata, this->which);
63  }
64 
65  /* handle discontinuity */
66  if (model->ne > 0 && rdata->nmaxevent > 0 && iroot >= 0) {
67  if (tnext == discs.at(iroot)) {
68  handleEventB(iroot);
69  --iroot;
70  }
71  }
72 
73  /* handle data-point */
74  if (tnext == rdata->ts[it]) {
75  handleDataPointB(it);
76  it--;
77  }
78 
79  /* reinit states */
80  solver->reInitB(which, t, xB, dxB);
81  solver->quadReInitB(which, xQB);
82  }
83 
84  /* we still need to integrate from first datapoint to tstart */
85  if (t > model->t0()) {
86  /* solve for backward problems */
87  solver->runB(model->t0());
88  solver->writeSolutionB(&t, xB, dxB, xQB, this->which);
89  solver->getDiagnosisB(0, rdata, this->which);
90  }
91 
92  computeLikelihoodSensitivities();
93  rdata->cpu_timeB = solver->getCpuTimeB();
94 }

149 void BackwardProblem::computeLikelihoodSensitivities()
150 {
151  for (int iJ = 0; iJ < model->nJ; iJ++) {
152  if (iJ == 0) {
153  for (int ip = 0; ip < model->nplist(); ++ip) {
154  llhS0[ip] = 0.0;
155  for (int ix = 0; ix < model->nxtrue_solver; ++ix) {
156  llhS0[ip] += xB[ix] * sx0.at(ix,ip);
157  }
158  }
159  } else {
160  for (int ip = 0; ip < model->nplist(); ++ip) {
161  llhS0[ip + iJ * model->nplist()] = 0.0;
162  for (int ix = 0; ix < model->nxtrue_solver; ++ix) {
163  llhS0[ip + iJ * model->nplist()] +=
164  xB[ix + iJ * model->nxtrue_solver] * sx0.at(ix,ip)+
165  xB[ix] * sx0.at(ix + iJ * model->nxtrue_solver,ip);
166  }
167  }
168  }
169  }
170 
171  for (int iJ = 0; iJ < model->nJ; iJ++) {
172  for (int ip = 0; ip < model->nplist(); ip++) {
173  if (iJ == 0) {
174  rdata->sllh.at(ip) -= llhS0[ip] + xQB[ip*model->nJ];
175  } else {
176  rdata->s2llh.at(iJ - 1 + ip * (model->nJ - 1)) -=
177  llhS0[ip + iJ * model->nplist()] +
178  xQB[iJ + ip*model->nJ];
179  }
180  }
