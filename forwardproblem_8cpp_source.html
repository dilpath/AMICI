53 void ForwardProblem::workForwardProblem() {
54  bool computeSensitivities =
55  solver->getSensitivityOrder() >= SensitivityOrder::first &&
56  model->nx_solver > 0;
57 
58  model->initialize(x, dx, sx, sdx, computeSensitivities);
59  solver->setup(model->t0(), model, x, dx, sx, sdx);
60  // update x0 after computing consistence IC, only important for DAEs
61  x.copy(solver->getState(model->t0()));
62 
63  model->fx_rdata(x_rdata, x);
64  if(solver->getSensitivityOrder() >= SensitivityOrder::first) {
65  model->fsx_rdata(sx_rdata, sx);
66  }
67 
68  if(edata){
69  rdata->initializeObjectiveFunction();
70  }
71 
72  /* if preequilibration is necessary, start Newton solver */
73  if (solver->getPreequilibration() || (edata && !edata->fixedParametersPreequilibration.empty())) {
74  handlePreequilibration();
75  } else {
76  model->fx_rdata(x_rdata, x);
77  rdata->x0 = x_rdata.getVector();
78  if (solver->getSensitivityMethod() == SensitivityMethod::forward &&
79  solver->getSensitivityOrder() >= SensitivityOrder::first) {
80  model->fsx_rdata(sx_rdata, sx);
81  for (int ix = 0; ix < rdata->nx; ix++) {
82  for (int ip = 0; ip < model->nplist(); ip++)
83  rdata->sx0[ip*rdata->nx + ix] = sx_rdata.at(ix,ip);
84  }
85  }
86  }
87 
88  /* perform presimulation if necessary */
89  if (edata && edata->t_presim > 0)
90  handlePresimulation();
91 
92  /* loop over timepoints */
93  for (int it = 0; it < model->nt(); it++) {
94  auto nextTimepoint = model->getTimepoint(it);
95 
96  if (nextTimepoint > model->t0()) {
97  if (model->nx_solver == 0) {
98  t = nextTimepoint;
99  break;
100  }
101 
102  // Solve for nextTimepoint
103  while (t < nextTimepoint) {
104  if (std::isinf(nextTimepoint)) {
105  SteadystateProblem sstate = SteadystateProblem(solver, x);
106  sstate.workSteadyStateProblem(rdata, solver, model, it);
107  sstate.writeSolution(&t, x, sx);
108  } else {
109  int status = solver->run(nextTimepoint);
110  solver->writeSolution(&t, x, dx, sx);
111  /* sx will be copied from solver on demand if sensitivities
112  are computed */
113  if (status == AMICI_ILL_INPUT) {
114  /* clustering of roots => turn off rootfinding */
115  solver->turnOffRootFinding();
116  } else if (status == AMICI_ROOT_RETURN) {
117  handleEvent(&tlastroot, false);
118  }
119  }
120  }
121  }
122  handleDataPoint(it);
123  }
124 
125  /* fill events */
126  if (model->nz > 0) {
127  getEventOutput();
128  }
129 
130  // set likelihood
131  if (!edata) {
132  rdata->invalidateLLH();
133  rdata->invalidateSLLH();
134  }
135 
136  storeJacobianAndDerivativeInReturnData();
137  rdata->cpu_time = solver->getCpuTime();
138 }

140 void ForwardProblem::handlePreequilibration() {
141  // Are there dedicated condition preequilibration parameters provided?
142  bool overrideFixedParameters =
143  edata && !edata->fixedParametersPreequilibration.empty();
144 
145  std::vector<realtype>
146  originalFixedParameters; // to restore after pre-equilibration
147 
148  if (overrideFixedParameters) {
149  if (edata->fixedParametersPreequilibration.size() !=
150  (unsigned)model->nk())
151  throw AmiException(
152  "Number of fixed parameters (%d) in model does not match "
153  "preequilibration parameters in ExpData (%zd).",
154  model->nk(), edata->fixedParametersPreequilibration.size());
155  originalFixedParameters = model->getFixedParameters();
156  model->setFixedParameters(edata->fixedParametersPreequilibration);
157  model->initialize(x, dx, sx, sdx,
158  solver->getSensitivityOrder() >=
159  SensitivityOrder::first);
160  }
161 
162  // pre-equilibrate
163  SteadystateProblem sstate = SteadystateProblem(solver, x);
164 
165  sstate.workSteadyStateProblem(rdata, solver, model, -1);
166  sstate.writeSolution(&t, x, sx);
167 
168  if (overrideFixedParameters) {
169  // Restore
170  model->setFixedParameters(originalFixedParameters);
171  }
172 
173  updateAndReinitStatesAndSensitivities(true);
174 }

208 void ForwardProblem::handlePresimulation()
209 {
210  // Are there dedicated condition preequilibration parameters provided?
211  bool overrideFixedParameters = edata && !edata->fixedParametersPresimulation.empty();
212 
213  std::vector<realtype> originalFixedParameters; // to restore after pre-equilibration
214 
215  if(overrideFixedParameters) {
216  if(edata->fixedParametersPresimulation.size() != (unsigned) model->nk())
217  throw AmiException("Number of fixed parameters (%d) in model does not match presimulation parameters in ExpData (%zd).",
218  model->nk(), edata->fixedParametersPresimulation.size());
219  originalFixedParameters = model->getFixedParameters();
220  model->setFixedParameters(edata->fixedParametersPresimulation);
221  }
222  t = model->t0() - edata->t_presim;
223  updateAndReinitStatesAndSensitivities(false);
224 
225  solver->run(model->t0());
226  solver->writeSolution(&t, x, dx, sx);
227  if(overrideFixedParameters) {
228  model->setFixedParameters(originalFixedParameters);
229  }
230  t = model->t0();
231  updateAndReinitStatesAndSensitivities(false);
232 }

235 void ForwardProblem::handleEvent(realtype *tlastroot, const bool seflag) {
236  /* store heaviside information at event occurence */
237  model->froot(t, x, dx, rootvals);
238 
239  if (!seflag) {
240  solver->getRootInfo(rootsfound.data());
241  }
242 
243  if (iroot < model->nMaxEvent() * model->ne) {
244  std::copy(rootsfound.begin(), rootsfound.end(),
245  &rootidx[iroot * model->ne]);
246  }
247 
248  rvaltmp = rootvals;
249 
250  if (!seflag) {
251  /* only extract in the first event fired */
252  if (solver->getSensitivityOrder() >= SensitivityOrder::first &&
253  solver->getSensitivityMethod() == SensitivityMethod::forward) {
254  sx.copy(solver->getStateSensitivity(t));
255  }
256 
257  /* only check this in the first event fired, otherwise this will always
258  * be true */
259  if (t == *tlastroot) {
260  throw AmiException("AMICI is stuck in an event, as the initial"
261  "step-size after the event is too small. To fix "
262  "this, increase absolute and relative tolerances!");
263  }
264  *tlastroot = t;
265  }
266 
267  if(model->nz>0)
268  getEventOutput();
269 
270  /* if we need to do forward sensitivities later on we need to store the old
271  * x and the old xdot */
272  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
273  /* store x and xdot to compute jump in sensitivities */
274  x_old = solver->getState(t);
275  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
276  model->fxdot(t, x, dx, xdot);
277  xdot_old = xdot;
278  dx_old = dx;
279 
280  /* compute event-time derivative only for primary events, we get
281  * into trouble with multiple simultaneously firing events here (but
282  * is this really well defined then?), in that case just use the
283  * last ie and hope for the best. */
284  if (!seflag) {
285  for (int ie = 0; ie < model->ne; ie++) {
286  if (rootsfound.at(ie) == 1) {
287  /* only consider transitions false -> true */
288  model->getEventTimeSensitivity(stau, t, ie, x, sx);
289  }
290  }
291  }
292  } else if (solver->getSensitivityMethod() == SensitivityMethod::adjoint) {
293  /* store x to compute jump in discontinuity */
294  if (iroot < model->nMaxEvent() * model->ne) {
295  x_disc[iroot] = x;
296  xdot_disc[iroot] = xdot;
297  xdot_old_disc[iroot] = xdot_old;
298  }
299  }
300  }
301 
302  model->updateHeaviside(rootsfound);
303 
304  applyEventBolus();
305 
306  if (iroot < model->nMaxEvent() * model->ne) {
307  discs[iroot] = t;
308  ++iroot;
309  } else {
310  solver->app->warning(
311  "AMICI:mex:TOO_MUCH_EVENT",
312  "Event was recorded but not reported as the number of "
313  "occured events exceeded (nmaxevents)*(number of "
314  "events in model definition)!");
315  /* reinitialise so that we can continue in peace */
316  solver->reInit(t, x, dx);
317  return;
318  }
319 
320  if (solver->getSensitivityOrder() >= SensitivityOrder::first
321  && solver->getSensitivityMethod() == SensitivityMethod::forward) {
322  /* compute the new xdot */
323  model->fxdot(t, x, dx, xdot);
324  applyEventSensiBolusFSA();
325  }
326 
327  int secondevent = 0;
328 
329  /* check whether we need to fire a secondary event */
330  model->froot(t, x, dx, rootvals);
331  for (int ie = 0; ie < model->ne; ie++) {
332  /* the same event should not trigger itself */
333  if (rootsfound.at(ie) == 0) {
334  /* check whether there was a zero-crossing */
335  if (0 > rvaltmp.at(ie) * rootvals.at(ie)) {
336  if (rvaltmp.at(ie) < rootvals.at(ie)) {
337  rootsfound.at(ie) = 1;
338  } else {
339  rootsfound.at(ie) = -1;
340  }
341  secondevent++;
342  } else {
343  rootsfound.at(ie) = 0;
344  }
345  } else {
346  /* don't fire the same event again */
347  rootsfound.at(ie) = 0;
348  }
349  }
350  /* fire the secondary event */
351  if (secondevent > 0) {
352  handleEvent(tlastroot, TRUE);
353  }
354 
355  /* only reinitialise in the first event fired */
356  if (!seflag) {
357  solver->reInit(t, x, dx);
358 
359  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
360  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
361  solver->sensReInit(sx, sdx);
362  }
363  }
364  }
365 }

381 void ForwardProblem::getEventOutput() {
382  if (t == model->getTimepoint(model->nt() - 1)) {
383  // call from fillEvent at last timepoint
384  model->froot(t, x, dx, rootvals);
385  }
386 
387  /* EVENT OUTPUT */
388  for (int ie = 0; ie < model->ne; ie++) {
389  /* only look for roots of the rootfunction not discontinuities */
390  if (nroots.at(ie) >= model->nMaxEvent())
391  continue;
392 
393  /* only consider transitions false -> true or event filling */
394  if (rootsfound.at(ie) != 1 &&
395  t != model->getTimepoint(model->nt() - 1)) {
396  continue;
397  }
398 
399  /* get event output */
400  model->getEvent(slice(rdata->z, nroots.at(ie), rdata->nz), ie, t, x);
401  /* if called from fillEvent at last timepoint,
402  then also get the root function value */
403  if (t == model->getTimepoint(model->nt() - 1))
404  model->getEventRegularization(slice(rdata->rz, nroots.at(ie),
405  rdata->nz), ie, t, x);
406 
407  if (edata) {
408  model->getEventSigma(slice(rdata->sigmaz, nroots.at(ie), rdata->nz),
409  ie, nroots.at(ie), t, edata);
410  model->addEventObjective(rdata->llh, ie, nroots.at(ie), t, x,
411  *edata);
412 
413  /* if called from fillEvent at last timepoint,
414  add regularization based on rz */
415  if (t == model->getTimepoint(model->nt() - 1))
416  model->addEventObjectiveRegularization(
417  rdata->llh, ie, nroots.at(ie), t, x, *edata);
418  }
419 
420  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
421  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
422  getEventSensisFSA(ie);
423  } else {
424  if (edata) {
425  model->getAdjointStateEventUpdate(slice(dJzdx, nroots.at(ie),
426  model->nx_solver * model->nJ),
427  ie, nroots.at(ie), t, x, *edata);
428  model->addPartialEventObjectiveSensitivity(rdata->sllh,
429  rdata->s2llh,
430  ie, nroots.at(ie),
431  t, x, *edata);
432  }
433  }
434  }
435 
436  nroots.at(ie)++;
437  }
438 
439  if (t == model->getTimepoint(model->nt() - 1)) {
440  // call from fillEvent at last timepoint
441  // loop until all events are filled
442  if (std::any_of(nroots.cbegin(), nroots.cend(), [&](int curNRoots) {
443  return curNRoots < model->nMaxEvent();
444  }))
445  getEventOutput();
446  }
447 }
