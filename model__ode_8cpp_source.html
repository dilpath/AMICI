352 void Model_ODE::fqBdot(realtype t, N_Vector x, N_Vector xB, N_Vector qBdot) {
353  /* initialize with zeros */
354  N_VConst(0.0, qBdot);
355  fdxdotdp(t, x);
356 
357  if (pythonGenerated) {
358  /* call multiplication */
359  if (ndxdotdp_explicit > 0)
360  dxdotdp_explicit.multiply(qBdot, xB, plist_, true);
361  if (ndxdotdp_implicit > 0)
362  dxdotdp_implicit.multiply(qBdot, xB, plist_, true);
363  N_VScale(-1.0, qBdot, qBdot);
364  } else {
365  /* was matlab generated */
366  for (int ip = 0; ip < nplist(); ip++) {
367  for (int ix = 0; ix < nxtrue_solver; ix++)
368  NV_Ith_S(qBdot, ip * nJ) -= NV_Ith_S(xB, ix) * dxdotdp.at(ix, ip);
369  // second order part
370  for (int iJ = 1; iJ < nJ; iJ++)
371  for (int ix = 0; ix < nxtrue_solver; ix++)
372  NV_Ith_S(qBdot, ip * nJ + iJ) -=
373  NV_Ith_S(xB, ix) * dxdotdp.at(ix + iJ * nxtrue_solver, ip) +
374  NV_Ith_S(xB, ix + iJ * nxtrue_solver) * dxdotdp.at(ix, ip);
375  }
376  }

386 void Model_ODE::fsxdot(realtype t, N_Vector x, int ip, N_Vector sx,
387  N_Vector sxdot) {
388 
389  /* sxdot is just the total derivative d(xdot)dp,
390  so we just call dxdotdp and copy the stuff over */
391  if (ip == 0) {
392  // we only need to call this for the first parameter index will be
393  // the same for all remaining
394  fdxdotdp(t, x);
395  fJSparse(t, x, J.get());
396  }
397  if (pythonGenerated) {
398  /* copy dxdotdp and the implicit version over */
399  // initialize
400  N_VConst(0.0, sxdot);
401  realtype *sxdot_tmp = N_VGetArrayPointer(sxdot);
402 
403  // copy explicit version
404  if (ndxdotdp_explicit > 0) {
405  auto col_exp = dxdotdp_explicit.indexptrs();
406  auto row_exp = dxdotdp_explicit.indexvals();
407  auto data_exp_ptr = dxdotdp_explicit.data();
408  for (sunindextype i = col_exp[plist(ip)]; i < col_exp[plist(ip) + 1]; ++i)
409  sxdot_tmp[row_exp[i]] += data_exp_ptr[i];
410  }
411 
412  // copy implicit version
413  if (ndxdotdp_implicit > 0) {
414  auto col_imp = dxdotdp_implicit.indexptrs();
415  auto row_imp = dxdotdp_implicit.indexvals();
416  auto data_imp_ptr = dxdotdp_implicit.data();
417  for (sunindextype i = col_imp[plist(ip)]; i < col_imp[plist(ip) + 1]; ++i)
418  sxdot_tmp[row_imp[i]] += data_imp_ptr[i];
419  }
420 
421  } else {
422  /* copy dxdotdp over */
423  N_VScale(1.0, dxdotdp.getNVector(ip), sxdot);
424  }
425  J.multiply(sxdot, sx);
426 }
