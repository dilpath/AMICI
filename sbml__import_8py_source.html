438  def processSpecies(self):

440

442  species_element.getId(): species_index
443  for species_index, species_element in enumerate(species)
444  }
445 
446  self.symbols['species']['identifier'] = sp.Matrix(
447  [sp.Symbol(spec.getId(), real=True) for spec in species]
448  )
449  self.symbols['species']['name'] = [spec.getName() for spec in species]

452  [sp.Symbol(spec.getCompartment(), real=True) for spec in species]
453  )

456  for species_element in species

465  specie.getHasOnlySubstanceUnits() for specie in species

469  amounts = [spec.getInitialAmount() for spec in species]

471  def get_species_initial(index, conc):
472  # We always simulate concentrations!
473  if self.speciesHasOnlySubstanceUnits[index]:
474  if species[index].isSetInitialAmount() \
475  and not math.isnan(amounts[index]):
476  return sp.sympify(amounts[index]) \
477  / self.speciesCompartment[index]
478  if species[index].isSetInitialConcentration():
479  return sp.sympify(conc)
480  else:
481  if species[index].isSetInitialConcentration():
482  return sp.sympify(conc)
483 
484  if species[index].isSetInitialAmount() \
485  and not math.isnan(amounts[index]):
486  return sp.sympify(amounts[index]) \
487  / self.speciesCompartment[index]
488 
489  return self.symbols['species']['identifier'][index]
490 
491  species_initial = sp.Matrix(
492  [get_species_initial(index, conc)
493  for index, conc in enumerate(concentrations)]
494  )
495 
496  species_ids = [spec.getId() for spec in self.sbml.getListOfSpecies()]
497  for initial_assignment in self.sbml.getListOfInitialAssignments():
498  if initial_assignment.getId() in species_ids:
499  index = species_ids.index(
500  initial_assignment.getId()
501  )
502  symMath = sp.sympify(_parse_logical_operators(
503  sbml.formulaToL3String(initial_assignment.getMath())),
504  locals=self.local_symbols
505  )
506  if symMath is not None:
507  symMath = _parse_special_functions(symMath)
508  _check_unsupported_functions(symMath, 'InitialAssignment')
509  species_initial[index] = symMath
510 
511  for ix, (symbol, init) in enumerate(zip(
512  self.symbols['species']['identifier'], species_initial
513  )):
514  if symbol == init:
515  species_initial[ix] = sp.sympify(0.0)
516 
517  # flatten initSpecies
518  while any([species in species_initial.free_symbols
519  for species in self.symbols['species']['identifier']]):
520  species_initial = species_initial.subs([
521  (symbol, init)
522  for symbol, init in zip(
523  self.symbols['species']['identifier'], species_initial
524  )
525  ])
526 
527  self.symbols['species']['value'] = species_initial
528 
529  if self.sbml.isSetConversionFactor():
530  conversion_factor = sp.Symbol(self.sbml.getConversionFactor(),
531  real=True)
532  else:
533  conversion_factor = 1.0
534 
535  self.speciesConversionFactor = sp.Matrix([
536  sp.sympify(specie.getConversionFactor())
537  if specie.isSetConversionFactor()
538  else conversion_factor
539  for specie in species
540  ])

659  @log_execution_time('processing SBML reactions', logger)
660  def processReactions(self):
661  reactions = self.sbml.getListOfReactions()
662  nr = len(reactions)
663  nx = len(self.symbols['species']['name'])
664  # stoichiometric matrix
665  self.stoichiometricMatrix = sp.SparseMatrix(sp.zeros(nx, nr))
666  self.fluxVector = sp.zeros(nr, 1)
667 
668  assignment_ids = [ass.getId()
669  for ass in self.sbml.getListOfInitialAssignments()]
670  rulevars = [rule.getVariable()
671  for rule in self.sbml.getListOfRules()
672  if rule.getFormula() != '']
673 
674  reaction_ids = [
675  reaction.getId() for reaction in reactions
676  if reaction.isSetId()
677  ]
678 
679  def getElementFromAssignment(element_id):
680  assignment = self.sbml.getInitialAssignment(
681  element_id
682  )
683  sym = sp.sympify(sbml.formulaToL3String(assignment.getMath()),
684  locals=self.local_symbols)
685  # this is an initial assignment so we need to use
686  # initial conditions
687  if sym is not None:
688  sym = sym.subs(
689  self.symbols['species']['identifier'],
690  self.symbols['species']['value']
691  )
692  return sym
693 
694  def getElementStoichiometry(element):
695  if element.isSetId():
696  if element.getId() in assignment_ids:
697  symMath = getElementFromAssignment(element.getId())
698  if symMath is None:
699  symMath = sp.sympify(element.getStoichiometry())
700  elif element.getId() in rulevars:
701  return sp.Symbol(element.getId(), real=True)
702  else:
703  # dont put the symbol if it wont get replaced by a
704  # rule
705  symMath = sp.sympify(element.getStoichiometry())
706  elif element.isSetStoichiometry():
707  symMath = sp.sympify(element.getStoichiometry())
708  else:
709  return sp.sympify(1.0)
710  symMath = _parse_special_functions(symMath)
711  _check_unsupported_functions(symMath, 'Stoichiometry')
712  return symMath
713 
714  def isConstant(specie):
715  return specie in self.constantSpecies or \
716  specie in self.boundaryConditionSpecies
717 
718  for reactionIndex, reaction in enumerate(reactions):
719  for elementList, sign in [(reaction.getListOfReactants(), -1.0),
720  (reaction.getListOfProducts(), 1.0)]:
721  elements = {}
722  for index, element in enumerate(elementList):
723  # we need the index here as we might have multiple elements
724  # for the same species
725  elements[index] = {'species': element.getSpecies()}
726  elements[index]['stoichiometry'] = getElementStoichiometry(
727  element
728  )
729 
730  for index in elements.keys():
731  if not isConstant(elements[index]['species']):
732  specieIndex = self.speciesIndex[
733  elements[index]['species']
734  ]
735  self.stoichiometricMatrix[specieIndex, reactionIndex] \
736  += sign \
737  * elements[index]['stoichiometry'] \
738  * self.speciesConversionFactor[specieIndex] \
739  / self.speciesCompartment[specieIndex]
740 
741  # usage of formulaToL3String ensures that we get "time" as time
742  # symbol
743  math = sbml.formulaToL3String(reaction.getKineticLaw().getMath())
744  try:
745  symMath = sp.sympify(_parse_logical_operators(math),
746  locals=self.local_symbols)
747  except SBMLException as Ex:
748  raise Ex
749  except:
750  raise SBMLException(f'Kinetic law "{math}" contains an '
751  'unsupported expression!')
752  symMath = _parse_special_functions(symMath)
753  _check_unsupported_functions(symMath, 'KineticLaw')
754  for r in reactions:
755  elements = list(r.getListOfReactants()) \
756  + list(r.getListOfProducts())
757  for element in elements:
758  if element.isSetId() & element.isSetStoichiometry():
759  symMath = symMath.subs(
760  sp.sympify(element.getId(),
761  locals=self.local_symbols),
762  sp.sympify(element.getStoichiometry())
763  )
764 
765  self.fluxVector[reactionIndex] = symMath
766  if any([
767  str(symbol) in reaction_ids
768  for symbol in self.fluxVector[reactionIndex].free_symbols
769  ]):
770  raise SBMLException(
771  'Kinetic laws involving reaction ids are currently'
772  ' not supported!'
773  )

784  @log_execution_time('processing SBML rules', logger)
785  def processRules(self):
786  rules = self.sbml.getListOfRules()
787 
788  rulevars = getRuleVars(rules, local_symbols=self.local_symbols)
789  fluxvars = self.fluxVector.free_symbols
790  specvars = self.symbols['species']['identifier'].free_symbols
791  volumevars = self.compartmentVolume.free_symbols
792  compartmentvars = self.compartmentSymbols.free_symbols
793  parametervars = sp.Matrix([
794  sp.Symbol(par.getId(), real=True)
795  for par in self.sbml.getListOfParameters()
796  ])
797  stoichvars = self.stoichiometricMatrix.free_symbols
798 
799  assignments = {}
800 
801  for rule in rules:
802  if rule.getFormula() == '':
803  continue
804  variable = sp.sympify(rule.getVariable(),
805  locals=self.local_symbols)
806  # avoid incorrect parsing of pow(x, -1) in symengine
807  formula = sp.sympify(_parse_logical_operators(
808  sbml.formulaToL3String(rule.getMath())),
809  locals=self.local_symbols)
810  formula = _parse_special_functions(formula)
811  _check_unsupported_functions(formula, 'Rule')
812 
813  if variable in stoichvars:
814  self.stoichiometricMatrix = \
815  self.stoichiometricMatrix.subs(variable, formula)
816 
817  if variable in specvars:
818  raise SBMLException('Species assignment rules are currently'
819  ' not supported!')
820 
821  if variable in compartmentvars:
822  raise SBMLException('Compartment assignment rules are'
823  ' currently not supported!')
824 
825  if variable in parametervars:
826  try:
827  idx = self.parameterIndex[str(variable)]
828  self.symbols['parameter']['value'][idx] \
829  = float(formula)
830  except:
831  self.sbml.removeParameter(str(variable))
832  assignments[str(variable)] = formula
833 
834  if variable in fluxvars:
835  self.fluxVector = self.fluxVector.subs(variable, formula)
836 
837  if variable in volumevars:
838  self.compartmentVolume = \
839  self.compartmentVolume.subs(variable, formula)
840 
841  if variable in rulevars:
842  for nested_rule in rules:
843  nested_formula = sp.sympify(
844  sbml.formulaToL3String(nested_rule.getMath()),
845  locals=self.local_symbols)
846  nested_formula = \
847  nested_formula.subs(variable, formula)
848  nested_rule.setFormula(str(nested_formula))
849 
850  for variable in assignments:
851  assignments[variable].subs(variable, formula)
852 
853  # do this at the very end to ensure we have flattened all recursive
854  # rules
855  for variable in assignments.keys():
856  self.replaceInAllExpressions(
857  sp.Symbol(variable, real=True),
858  assignments[variable]
859  )
860  for comp, vol in zip(self.compartmentSymbols, self.compartmentVolume):
861  self.replaceInAllExpressions(
862  comp, vol
863  )

1174 def replaceLogAB(x):
1175 
1176  match = re.search(r'(^|\W)log\(', x)
1177  if not match:
1178  return x
1179 
1180  # index of 'l' of 'log'
1181  logStart = match.start() \
1182  if match.end() - match.start() == 4 \
1183  else match.start() + 1
1184  level = 0 # parenthesis level
1185  posComma = -1 # position of comma in log(a,b)
1186  for i in range(logStart + 4, len(x)):
1187  if x[i] == '(':
1188  level += 1
1189  elif x[i] == ')':
1190  level -= 1
1191  if level == -1: break
1192  elif x[i] == ',' and level == 0:
1193  posComma = i
1194 
1195  if posComma < 0:
1196  # was log(a), not log(a,b), so nothing to replace
1197  return x
1198 
1199  prefix = x[:logStart]
1200  suffix = x[i+1:]
1201  basis = x[logStart+4: posComma]
1202  a = x[posComma+1: i]
1203 
1204  replacement = f'{prefix}ln({a})/ln({basis}){suffix}'
1205 
1206  return replaceLogAB(replacement)
