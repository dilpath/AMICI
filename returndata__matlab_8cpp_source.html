186 void writeMatlabField2(mxArray *matlabStruct, const char *fieldName,
187  std::vector<T> const& fieldData, int dim0, int dim1,
188  std::vector<int> perm) {
189  if(fieldData.size() != dim0*dim1)
190  throw AmiException("Dimension mismatch when writing rdata->%s to matlab results",fieldName);
191 
192  if(perm.size() != 2)
193  throw AmiException("Dimension mismatch when applying permutation!");
194 
195  std::vector<mwSize> dim = {(mwSize)(dim0), (mwSize)(dim1)};
196 
197  double *array = initAndAttachArray(matlabStruct, fieldName, reorder(dim,perm));
198 
199  std::vector<int> index = {0,0};
200  /* transform rowmajor (c++) to colmajor (matlab) and apply permutation */
201  for (index[0] = 0; index[0] < dim[0]; index[0]++) {
202  for (index[1] = 0; index[1] < dim[1]; index[1]++) {
203  array[index[perm[0]] + index[perm[1]]*dim[perm[0]]] =
204  static_cast<double>(fieldData[index[0]*dim[1] + index[1]]);
205  }
206  }

263 double *initAndAttachArray(mxArray *matlabStruct, const char *fieldName, std::vector<mwSize> dim) {
264  if(!mxIsStruct(matlabStruct))
265  throw AmiException("Passing non-struct mxArray to initAndAttachArray!",fieldName);
266 
267  int fieldNumber = mxGetFieldNumber(matlabStruct, fieldName);
268  if(fieldNumber<0)
269  throw AmiException("Trying to access non-existent field '%s'!",fieldName);
270 
271  mxArray *array = mxCreateNumericArray(dim.size(), dim.data(), mxDOUBLE_CLASS, mxREAL);
272  mxSetFieldByNumber(matlabStruct, 0, fieldNumber, array);
273  return(mxGetPr(array));
274 }
