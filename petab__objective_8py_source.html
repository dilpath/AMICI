557 def _get_timepoints_with_replicates(
558  df_for_condition: pd.DataFrame) -> List[numbers.Number]:
559  # create sorted list of all timepoints for which measurements exist
560  timepoints = sorted(df_for_condition[TIME].unique().astype(float))
561 
562  # find replicate numbers of time points
563  timepoints_w_reps = []
564  for time in timepoints:
565  # subselect for time
566  df_for_time = df_for_condition[df_for_condition.time == time]
567  # rep number is maximum over rep numbers for observables
568  n_reps = max(df_for_time.groupby(
569  [OBSERVABLE_ID, TIME]).size())
570  # append time point n_rep times
571  timepoints_w_reps.extend([time] * n_reps)
572 
573  return timepoints_w_reps

591 def _get_measurements_and_sigmas(
592  df_for_condition: pd.DataFrame,
593  timepoints_w_reps: Sequence[numbers.Number],
594  observable_ids: Sequence[str]) -> Tuple[np.array, np.array]:
595  # prepare measurement matrix
596  y = np.full(shape=(len(timepoints_w_reps), len(observable_ids)),
597  fill_value=np.nan)
598  # prepare sigma matrix
599  sigma_y = y.copy()
600 
601  timepoints = sorted(df_for_condition[TIME].unique().astype(float))
602 
603  for time in timepoints:
604  # subselect for time
605  df_for_time = df_for_condition[df_for_condition[TIME] == time]
606  time_ix_0 = timepoints_w_reps.index(time)
607 
608  # remember used time indices for each observable
609  time_ix_for_obs_ix = {}
610 
611  # iterate over measurements
612  for _, measurement in df_for_time.iterrows():
613  # extract observable index
614  observable_ix = observable_ids.index(measurement[OBSERVABLE_ID])
615 
616  # update time index for observable
617  if observable_ix in time_ix_for_obs_ix:
618  time_ix_for_obs_ix[observable_ix] += 1
619  else:
620  time_ix_for_obs_ix[observable_ix] = time_ix_0
621 
622  # fill observable and possibly noise parameter
623  y[time_ix_for_obs_ix[observable_ix],
624  observable_ix] = measurement[MEASUREMENT]
625  if isinstance(measurement[NOISE_PARAMETERS], numbers.Number):
626  sigma_y[time_ix_for_obs_ix[observable_ix],
627  observable_ix] = measurement[NOISE_PARAMETERS]
628  return y, sigma_y

647 def rdatas_to_measurement_df(
648  rdatas: Sequence[amici.ReturnData],
649  model: amici.Model,
650  measurement_df: pd.DataFrame) -> pd.DataFrame:
651 
652  df = pd.DataFrame(columns=list(measurement_df.columns))
653 
654  simulation_conditions = petab.get_simulation_conditions(
655  measurement_df)
656 
657  observable_ids = model.getObservableIds()
658 
659  # iterate over conditions
660  for (_, condition), rdata in zip(simulation_conditions.iterrows(), rdatas):
661  # current simulation matrix
662  y = rdata['y']
663  # time array used in rdata
664  t = list(rdata['t'])
665

667  cur_measurement_df = petab.get_rows_for_condition(
668  measurement_df, condition)
669 
670  # iterate over entries for the given condition
671  # note: this way we only generate a dataframe entry for every
672  # row that existed in the original dataframe. if we want to
673  # e.g. have also timepoints non-existent in the original file,
674  # we need to instead iterate over the rdata['y'] entries
675  for _, row in cur_measurement_df.iterrows():
676  # copy row
677  row_sim = copy.deepcopy(row)
678 
679  # extract simulated measurement value
680  timepoint_idx = t.index(row[TIME])
681  observable_idx = observable_ids.index(row[OBSERVABLE_ID])
682  measurement_sim = y[timepoint_idx, observable_idx]
683 
684  # change measurement entry
685  row_sim.measurement = measurement_sim
686 
687  # append to dataframe
688  df = df.append(row_sim, ignore_index=True)

705 def aggregate_sllh(
706  amici_model: amici.Model,
707  rdatas: Sequence[amici.ReturnDataView],
708  parameter_mapping: Optional[List[petab.ParMappingDictTuple]],
709 ) -> Union[None, Dict[str, float]]:
710  sllh = {}
711  model_par_ids = amici_model.getParameterIds()
712  for (_, par_map_sim), rdata in zip(parameter_mapping, rdatas):
713  if rdata['status'] != amici.AMICI_SUCCESS \
714  or 'sllh' not in rdata\
715  or rdata['sllh'] is None:
716  return None
717 
718  for model_par_id, problem_par_id in par_map_sim.items():
719  if isinstance(problem_par_id, str):
720  model_par_idx = model_par_ids.index(model_par_id)
721  cur_par_sllh = rdata['sllh'][model_par_idx]
722  try:
723  sllh[problem_par_id] += cur_par_sllh
724  except KeyError:
725  sllh[problem_par_id] = cur_par_sllh
726  return sllh
