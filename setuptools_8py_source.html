30 def getBlasConfig():
31 
32  blaspkgcfg = {'include_dirs': [],
33  'library_dirs': [],
34  'libraries': [],
35  'define_macros': [],
36  'extra_compile_args': [],
37  'extra_link_args': []
38  }
39 
40  # Check environment variables
41  if 'BLAS_CFLAGS' in os.environ:
42  blaspkgcfg['extra_compile_args'].extend(
43  shlex.split(os.environ['BLAS_CFLAGS'])
44  )
45 
46  if 'BLAS_LIBS' in os.environ:
47  blaspkgcfg['extra_link_args'].extend(
48  shlex.split(os.environ['BLAS_LIBS'])
49  )
50 
51  if 'BLAS_CFLAGS' in os.environ or 'BLAS_LIBS' in os.environ:
52  # If options have been provided by the user, we don't try to detect
53  # anything by ourselves
54  return blaspkgcfg
55 
56  # Try environment modules
57  # MKL
58  if 'MKLROOT' in os.environ:
59  if 'MKL_INC' in os.environ:
60  blaspkgcfg['extra_compile_args'].extend(
61  shlex.split(os.environ['MKL_INC'])
62  )
63  if 'MKL_LIB' in os.environ:
64  blaspkgcfg['extra_link_args'].extend(
65  shlex.split(os.environ['MKL_LIB'])
66  )
67  blaspkgcfg['define_macros'].append(('AMICI_BLAS_MKL', None),)
68  return blaspkgcfg

95 def getHdf5Config():
96  if pkgconfig:
97  h5pkgcfg = {}
98  try:
99  h5pkgcfg = pkgconfig.parse('hdf5')
100  except pkgconfig.PackageNotFoundError:
101  pass
102  # NOTE: Cannot use pkgconfig.exists('hdf5f'), since this is true
103  # although no libraries or include dirs are available
104  h5pkgcfg['found'] = 'include_dirs' in h5pkgcfg \
105  and h5pkgcfg['include_dirs']
106  if h5pkgcfg['found']:
107  return h5pkgcfg
108 
109  h5pkgcfg = {'include_dirs': [],
110  'library_dirs': [],
111  'libraries': [],
112  'define_macros': []
113  }
114 
115  # try for hdf5 in standard locations
116  hdf5_include_dir_hints = [
117  '/usr/include/hdf5/serial',
118  '/usr/local/include',
119  '/usr/include', # travis ubuntu xenial, centos
120  '/usr/local/Cellar/hdf5/1.10.2_1/include' # travis macOS
121  ]
122  hdf5_library_dir_hints = [
123  '/usr/lib/x86_64-linux-gnu/', # travis ubuntu xenial
124  '/usr/lib/x86_64-linux-gnu/hdf5/serial',
125  '/usr/local/lib',
126  '/usr/lib64/', # CentOS
127  '/usr/local/Cellar/hdf5/1.10.2_1/lib' # travis macOS
128  ]
129 
130  # Check for Environment Modules variables
131  if 'HDF5_BASE' in os.environ:
132  hdf5_include_dir_hints.insert(
133  0, os.path.join(os.environ['HDF5_BASE'], 'include'))
134  hdf5_library_dir_hints.insert(
135  0, os.path.join(os.environ['HDF5_BASE'], 'lib'))
136 
137  for hdf5_include_dir_hint in hdf5_include_dir_hints:
138  hdf5_include_dir_found = os.path.isfile(
139  os.path.join(hdf5_include_dir_hint, 'hdf5.h'))
140  if hdf5_include_dir_found:
141  print('hdf5.h found in %s' % hdf5_include_dir_hint)
142  h5pkgcfg['include_dirs'] = [hdf5_include_dir_hint]
143  break
144 
145  for hdf5_library_dir_hint in hdf5_library_dir_hints:
146  # check for static or shared library
147  for lib_filename in ['libhdf5.a', 'libhdf5.so']:
148  hdf5_library_dir_found = os.path.isfile(
149  os.path.join(hdf5_library_dir_hint, lib_filename))
150  if hdf5_library_dir_found:
151  print(f'{lib_filename} found in {hdf5_library_dir_hint}')
152  h5pkgcfg['library_dirs'] = [hdf5_library_dir_hint]
153  break
154  if hdf5_library_dir_found:
155  # break to not override hdf5_library_dir_found
156  break
157  h5pkgcfg['found'] = hdf5_include_dir_found and hdf5_library_dir_found
158 
159  return h5pkgcfg

205 def generateSwigInterfaceFiles():
206  swig_outdir = '%s/amici' % os.path.abspath(os.getcwd())
207  swig_exe = find_swig()
208  swig_version = get_swig_version(swig_exe)
209 
210  print(f"Found SWIG version {swig_version}")
211 
212  # Swig AMICI interface without HDF5 dependency
213  swig_cmd = [swig_exe,
214  '-c++',
215  '-python',
216  '-threads',
217  '-Iamici/swig', '-Iamici/include',
218  '-DAMICI_SWIG_WITHOUT_HDF5',
219  '-outdir', swig_outdir,
220  '-o', 'amici/amici_wrap_without_hdf5.cxx',
221  'amici/swig/amici.i']
222 
223  # Do we have -doxygen?
224  if swig_version >= (4, 0, 0):
225  swig_cmd.insert(1, '-doxygen')
226 
227  print(f"Running SWIG: {' '.join(swig_cmd)}")
228  sp = subprocess.run(swig_cmd, stdout=subprocess.PIPE,
229  stderr=sys.stdout.buffer)
230  if not sp.returncode == 0:
231  raise AssertionError('Swigging AMICI failed:\n'
232  + sp.stdout.decode('utf-8'))
233  shutil.move(os.path.join(swig_outdir, 'amici.py'),
234  os.path.join(swig_outdir, 'amici_without_hdf5.py'))
235 
236  # Swig AMICI interface with HDF5 dependency
237  swig_cmd = [swig_exe,
238  '-c++',
239  '-python',
240  '-threads',
241  '-Iamici/swig', '-Iamici/include',
242  '-outdir', swig_outdir,
243  '-o', 'amici/amici_wrap.cxx',
244  'amici/swig/amici.i']
245  if swig_version >= (4, 0, 0):
246  swig_cmd.insert(1, '-doxygen')
247  print(f"Running SWIG: {' '.join(swig_cmd)}")
