222 void Model_DAE::fqBdot(realtype t, N_Vector x, N_Vector dx, N_Vector xB,
223  N_Vector /*dxB*/, N_Vector qBdot) {
224  N_VConst(0.0, qBdot);
225  fdxdotdp(t, x, dx);
226  for (int ip = 0; ip < nplist(); ip++) {
227  for (int ix = 0; ix < nxtrue_solver; ix++)
228  NV_Ith_S(qBdot, ip * nJ) -= NV_Ith_S(xB, ix) * dxdotdp.at(ix, ip);
229  // second order part
230  for (int iJ = 1; iJ < nJ; iJ++)
231  for (int ix = 0; ix < nxtrue_solver; ix++)
232  NV_Ith_S(qBdot, ip * nJ + iJ) -=
233  NV_Ith_S(xB, ix) * dxdotdp.at(ix + iJ * nxtrue_solver, ip) +
234  NV_Ith_S(xB, ix + iJ * nxtrue_solver) * dxdotdp.at(ix, ip);
235  }
236 }

245 void Model_DAE::fsxdot(realtype t, N_Vector x, N_Vector dx, int ip, N_Vector sx,
246  N_Vector sdx, N_Vector sxdot) {
247  if (ip == 0) {
248  // we only need to call this for the first parameter index will be
249  // the same for all remaining
250  fM(t, x);
251  fdxdotdp(t, x, dx);
252  fJSparse(t, 0.0, x, dx, J.get());
253  }
254 
255  if (pythonGenerated) {
256  // python generated, not yet implemented for DAEs
257  throw AmiException("Wrapping of DAEs is not yet implemented from Python");
258  } else {
259  /* copy dxdotdp over */
260  N_VScale(1.0, dxdotdp.getNVector(ip), sxdot);
261  }
262 
263  J.multiply(sxdot, sx);
264  N_VScale(-1.0, sdx, sdx);
265  M.multiply(sxdot, sdx);
266  N_VScale(-1.0, sdx, sdx);
267 }
