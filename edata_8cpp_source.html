58 ExpData::ExpData(ReturnData const& rdata, std::vector<realtype> sigma_y,
59  std::vector<realtype> sigma_z)
60  : ExpData(rdata.nytrue, rdata.nztrue, rdata.nmaxevent, rdata.ts)
61 {
62  if (sigma_y.size() != (unsigned) nytrue_ && sigma_y.size() != (unsigned) nytrue_*nt())
63  throw AmiException("Dimension of sigma_y must be %d or %d, was %d", nytrue_, nytrue_*nt(), sigma_y.size());
64 
65  if (sigma_z.size() != (unsigned) nztrue_ && sigma_z.size() != (unsigned) nztrue_*nmaxevent_)
66  throw AmiException("Dimension of sigma_z must be %d or %d, was %d", nztrue_, nztrue_*nmaxevent_, sigma_z.size());
67 
68  std::random_device rd{};
69  std::mt19937 gen{rd()};
70 
71  realtype sigma;
72 
73  checkSigmaPositivity(sigma_y, "sigma_y");
74  checkSigmaPositivity(sigma_z, "sigma_z");
75 
76  for (int iy = 0; iy < nytrue_; ++iy) {
77  for (int it = 0; it < nt(); ++it) {
78  sigma = sigma_y.size() == (unsigned) nytrue_ ? sigma_y.at(iy) : sigma_y.at(iy + nytrue_ * it);
79  std::normal_distribution<> e{0, sigma};
80  observedData.at(iy + nytrue_ * it) = rdata.y.at(iy + rdata.ny * it) + e(gen);
81  observedDataStdDev.at(iy + nytrue_ * it) = sigma;
82  }
83  }
84 
85  for (int iz = 0; iz < nztrue_; ++iz) {
86  for (int ie = 0; ie < nmaxevent_; ++ie) {
87  sigma = sigma_z.size() == (unsigned) nztrue_ ? sigma_z.at(iz) : sigma_z.at(iz + nztrue_ * ie);
88  std::normal_distribution<> e{0, sigma};
89  observedEvents.at(iz + rdata.nztrue * ie) = rdata.z.at(iz + rdata.nz * ie) + e(gen);
90  observedDataStdDev.at(iz + rdata.nztrue * ie) = sigma;
91  }
92  }

345 void ConditionContext::applyCondition(const ExpData *edata)
346 {
347  if(!edata)
348  return;
349 
350  // this needs to go first, otherwise nplist will not have the right
351  // dimension for all other fields that depend on Model::nplist
352  if(!edata->plist.empty())
353  model->setParameterList(edata->plist);
354 
355  // this needs to go second as setParameterScale will reset sx0
356  if(!edata->pscale.empty()) {
357  if(edata->pscale.size() != (unsigned) model->np())
358  throw AmiException("Number of parameters (%d) in model does not"
359  " match ExpData (%zd).",
360  model->np(), edata->pscale.size());
361  model->setParameterScale(edata->pscale);
362 
363  }
364 
365  if(!edata->x0.empty()) {
366  if(edata->x0.size() != (unsigned) model->nx_rdata)
367  throw AmiException("Number of initial conditions (%d) in model does"
368  " not match ExpData (%zd).",
369  model->nx_rdata, edata->x0.size());
370  model->setInitialStates(edata->x0);
371  }
372 
373  if(!edata->sx0.empty()) {
374  if(edata->sx0.size() != (unsigned) model->nx_rdata * model->nplist())
375  throw AmiException("Number of initial conditions sensitivities (%d)"
376  " in model does not match ExpData (%zd).",
377  model->nx_rdata * model->nplist(),
378  edata->sx0.size());
379  model->setInitialStateSensitivities(edata->sx0);
380  }
381 
382  if(!edata->parameters.empty()) {
383  if(edata->parameters.size() != (unsigned) model->np())
384  throw AmiException("Number of parameters (%d) in model does not"
385  " match ExpData (%zd).",
386  model->np(), edata->parameters.size());
387  model->setParameters(edata->parameters);
388  }
389 
390  if(!edata->fixedParameters.empty()) {
391  // fixed parameter in model are superseded by those provided in edata
392  if(edata->fixedParameters.size() != (unsigned) model->nk())
393  throw AmiException("Number of fixed parameters (%d) in model does"
394  " not match ExpData (%zd).",
395  model->nk(), edata->fixedParameters.size());
396  model->setFixedParameters(edata->fixedParameters);
397  }
398 
399  if(edata->nt()) {
400  // fixed parameter in model are superseded by those provided in edata
401  model->setTimepoints(edata->getTimepoints());
402  }
403 }
