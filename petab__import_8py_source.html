47 def get_fixed_parameters(
48  sbml_model: 'libsbml.Model',
49  condition_df: Optional[pd.DataFrame] = None,
50  const_species_to_parameters: bool = False) -> List[str]:
51 
52  # Column names are model parameter IDs, compartment IDs or species IDs.
53  # Thereof, all parameters except for any overridden ones should be made
54  # constant.
55  # (Could potentially still be made constant, but leaving them might
56  # increase model reusability)
57 
58  # handle parameters in condition table
59  if condition_df is not None:
60  fixed_parameters = list(condition_df.columns)
61  # get rid of conditionName column
62  try:
63  fixed_parameters.remove(CONDITION_NAME)
64  except ValueError:
65  pass
66 
67  logger.debug(f'Condition table: {condition_df.shape}')
68 
69  # remove overridden parameters (`object`-type columns)
70  fixed_parameters = [p for p in fixed_parameters
71  if condition_df[p].dtype != 'O'
72  and sbml_model.getParameter(p) is not None]
73  # must be unique
74  if len(fixed_parameters) != len(set(fixed_parameters)):
75  raise AssertionError(
76  'len(fixed_parameters) != len(set(fixed_parameters))')
77  else:
78  fixed_parameters = []
79 
80  # Others are optional
81  if const_species_to_parameters:
82  # Turn species which are marked constant in the SBML model into
83  # parameters
84  constant_species = constant_species_to_parameters(sbml_model)
85 
86  logger.debug("Constant species converted to parameters: "
87  + str(len(constant_species)))
88  logger.info("Non-constant species "
89  + str(len(sbml_model.getListOfSpecies())))
90 
91  # ... and append them to the list of fixed_parameters
92  for species in constant_species:
93  if species not in fixed_parameters:
94  fixed_parameters.append(species)
95 
96  # Ensure mentioned parameters exist in the model. Remove additional ones
97  # from list
98  for fixed_parameter in fixed_parameters[:]:
99  # check global parameters
100  if not sbml_model.getParameter(fixed_parameter) \
101  and not sbml_model.getSpecies(fixed_parameter):
102  logger.warning(f"Parameter or species '{fixed_parameter}'"
103  " provided in condition table but not present in"
104  " model. Ignoring.")
105  fixed_parameters.remove(fixed_parameter)
106 
107  if condition_df is None:
108  return fixed_parameters
109 
110  # initial concentrations for species or initial compartment sizes in
111  # condition table will need to be turned into fixed parameters
112 
113  # if there is no initial assignment for that species, we'd need
114  # to create one. to avoid any naming collision right away, we don't allow
115  # that for now
116 
117  # we can't handle them yet
118  compartments = [col for col in condition_df
119  if sbml_model.getCompartment(col) is not None]
120  if compartments:
121  raise NotImplementedError("Can't handle initial compartment sizes "
122  "at the moment. Consider creating an "
123  f"initial assignment for {compartments}")
124 
125  species = [col for col in condition_df
126  if not np.issubdtype(condition_df[col].dtype, np.number)
127  and sbml_model.getSpecies(col) is not None]
128  if species:
129  raise NotImplementedError(
130  "Can't handle parameterized initial concentrations in condition "
131  f"table. Consider creating an initial assignment for {species}")
132 
133  return fixed_parameters

147 def species_to_parameters(species_ids: List[str],
148  sbml_model: 'libsbml.Model') -> List[str]:
149  transformables = []
150 
151  for species_id in species_ids:
152  species = sbml_model.getSpecies(species_id)
153 
154  if species.getHasOnlySubstanceUnits():
155  logger.warning(
156  f"Ignoring {species.getId()} which has only substance units."
157  " Conversion not yet implemented.")
158  continue
159 
160  if math.isnan(species.getInitialConcentration()):
161  logger.warning(
162  f"Ignoring {species.getId()} which has no initial "
163  "concentration. Amount conversion not yet implemented.")
164  continue
165 
166  transformables.append(species_id)
167 
168  # Must not remove species while iterating over getListOfSpecies()
169  for species_id in transformables:
170  species = sbml_model.removeSpecies(species_id)
171  par = sbml_model.createParameter()
172  par.setId(species.getId())
173  par.setName(species.getName())
174  par.setConstant(True)
175  par.setValue(species.getInitialConcentration())
176  par.setUnits(species.getUnits())
177 
178  # Remove from reactants and products
179  for reaction in sbml_model.getListOfReactions():
180  for species_id in transformables:
181  # loop, since removeX only removes one instance
182  while reaction.removeReactant(species_id):
183  pass
184  while reaction.removeProduct(species_id):
185  pass
186  while reaction.removeModifier(species_id):
187  pass
188 
189  return transformables

374 @log_execution_time('Importing PEtab model', logger)
375 def import_model(sbml_model: Union[str, 'libsbml.Model'],
376  condition_table: Optional[Union[str, pd.DataFrame]] = None,
377  observable_table: Optional[Union[str, pd.DataFrame]] = None,
378  model_name: Optional[str] = None,
379  model_output_dir: Optional[str] = None,
380  verbose: bool = True,
381  allow_reinit_fixpar_initcond: bool = True,
382  **kwargs) -> None:
383  if verbose:
384  logger.setLevel(verbose)
385 
386  logger.info(f"Importing model ...")
387 
388  # Get PEtab tables
389  observable_df = petab.get_observable_df(observable_table)
390  # to determine fixed parameters
391  condition_df = petab.get_condition_df(condition_table)
392 
393  if observable_df is None:
394  raise NotImplementedError("PEtab import without observables table "
395  "is currently not supported.")
396 
397  # Model name from SBML ID or filename
398  if model_name is None:
399  if isinstance(sbml_model, libsbml.Model):
400  model_name = sbml_model.getId()
401  else:
402  model_name = os.path.splitext(os.path.split(sbml_model)[-1])[0]
403 
404  if model_output_dir is None:
405  model_output_dir = os.path.join(os.getcwd(), model_name)
406 
407  logger.info(f"Model name is '{model_name}'. "
408  f"Writing model code to '{model_output_dir}'.")
409 
410  # Load model
411  if isinstance(sbml_model, str):
412  # from file
413  sbml_reader = libsbml.SBMLReader()
414  sbml_doc = sbml_reader.readSBMLFromFile(sbml_model)
415  sbml_model = sbml_doc.getModel()
416  else:
417  # Create a copy, because it will be modified by SbmlImporter
418  sbml_doc = sbml_model.getSBMLDocument().clone()
419  sbml_model = sbml_doc.getModel()
420 
421  show_model_info(sbml_model)
422 
423  sbml_importer = amici.SbmlImporter(sbml_model)
424  sbml_model = sbml_importer.sbml
425 
426  if observable_df is not None:
427  observables, noise_distrs, sigmas = \
428  get_observation_model(observable_df)
429 
430  logger.info(f'Observables: {len(observables)}')
431  logger.info(f'Sigmas: {len(sigmas)}')
432 
433  if not len(sigmas) == len(observables):
434  raise AssertionError(
435  f'Number of provided observables ({len(observables)}) and sigmas '
436  f'({len(sigmas)}) do not match.')
437 
438  # TODO: adding extra output parameters is currently not supported,
439  # so we add any output parameters to the SBML model.
440  # this should be changed to something more elegant
441  # <BeginWorkAround>
442  formulas = {val['formula'] for val in observables.values()}
443  formulas |= set(sigmas.values())
444  output_parameters = set()
445  for formula in formulas:
446  for free_sym in sp.sympify(formula).free_symbols:
447  sym = str(free_sym)
448  if sbml_model.getElementBySId(sym) is None:
449  output_parameters.add(sym)
450  logger.debug(f"Adding output parameters to model: {output_parameters}")
451  for par in output_parameters:
452  petab.add_global_parameter(sbml_model, par)
453  # <EndWorkAround>
454 
455  fixed_parameters = get_fixed_parameters(sbml_model=sbml_model,
456  condition_df=condition_df)
457 
458  logger.debug(f"Fixed parameters are {fixed_parameters}")
459  logger.info(f"Overall fixed parameters: {len(fixed_parameters)}")
460  logger.info("Variable parameters: "
461  + str(len(sbml_model.getListOfParameters())
462  - len(fixed_parameters)))
463 
464  # Create Python module from SBML model
465  sbml_importer.sbml2amici(
466  modelName=model_name,
467  output_dir=model_output_dir,
468  observables=observables,
469  constantParameters=fixed_parameters,
470  sigmas=sigmas,
471  allow_reinit_fixpar_initcond=allow_reinit_fixpar_initcond,
472  noise_distributions=noise_distrs,
473  verbose=verbose,
474  **kwargs)

478 # table in a format suitable for `sbml2amici`.

487 def get_observation_model(observable_df: pd.DataFrame
488  ) -> Tuple[Dict[str, Dict[str, str]],
489  Dict[str, str],
490  Dict[str, Union[str, float]]]:
491 
492  if observable_df is None:
493  return {}, {}, {}
494 
495  observables = {}
496  sigmas = {}
497 
498  for _, observable in observable_df.iterrows():
499  oid = observable.name
500  name = observable.get(OBSERVABLE_NAME, "")
501  formula_obs = observable[OBSERVABLE_FORMULA]
502  formula_noise = observable[NOISE_FORMULA]
503  observables[oid] = {'name': name, 'formula': formula_obs}
504  sigmas[oid] = formula_noise
505 
506  # Replace observableIds occurring in error model definition
507  for observable_id, formula in sigmas.items():
508  repl = sp.sympify(formula).subs(
509  observable_id, observables[observable_id]['formula'])
510  sigmas[observable_id] = str(repl)
511 
512  noise_distrs = petab_noise_distributions_to_amici(observable_df)
513 
514  return observables, noise_distrs, sigmas

528  amici_distrs = {}

530  amici_val = ''
531 
532  if OBSERVABLE_TRANSFORMATION in observable \

535  amici_val += observable[OBSERVABLE_TRANSFORMATION] + '-'

538  and isinstance(observable[NOISE_DISTRIBUTION], str) \

540  amici_val += observable[NOISE_DISTRIBUTION]
541  else:
542  amici_val += 'normal'
543  amici_distrs[observable.name] = amici_val
544
